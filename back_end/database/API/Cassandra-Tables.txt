CREATE TABLE cassandra.POST(
    postID int PRIMARY KEY,
    title text,
    authorID int,
    imagePaths list<text>,
    caption text,
    likes set<int>,
    date_posted TIMESTAMP,
    comments set<int>,
    numberLikes int,
    numberComments int
);

CREATE TABLE cassandra.DMTABLE(
    user1 int,
    user2 int,
    messageChunks list<int>,
    PRIMARY KEY (user1, user2)
);

CREATE TABLE cassandra.DMSUBSET(
    subsetID int PRIMARY KEY,
    messages list<text>,
    senders list<int>,
    time_sent list<TIMESTAMP>
);

CREATE TABLE cassandra.COMMENT(
    commentID int PRIMARY KEY,
    commenter int,
    content text,
    time TIMESTAMP
);

CREATE TABLE cassandra.WORMHOLE(
    user1 int,
    user2 int,
    messages list<text>,
    PRIMARY KEY (user1, user2)
);
