CREATE TABLE Post(
    postID int PRIMARY KEY
    title varchar(20),
    authourID int,
    imagePaths list<varchar(60)>,
    caption varchar(2000),
    likes set<int>,
    date_posted TIMESTAMP,
    comments set<int>,
    numberLikes int,
    numberComments int
)

CREATE TABLE DMTABLE(
    user1 int,
    user2 int,
    messageChunks list<int>,
    PRIMARY KEY (user1, user2)
)

CREATE TABLE DMSUBSET(
    subsetID int PRIMARY KEY,
    messages list<varchar(1000)>,
    senders list<int>,
    time_sent list<TIMESTAMP>,
)
CREATE TABLE COMMENT(
    commentID int PRIMARY KEY,
    commenter int,
    content varchar(400),
    time TIMESTAMP,
)

CREATE TABLE WORMHOLE(
    user1 int,
    user2 int,
    messages list<varchar(1000)>,
)
